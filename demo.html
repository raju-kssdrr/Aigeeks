// Upserting to Pinecone
async function upsertEmbeddings(index, vectors) {
  const upsertRequest = {
    vectors: vectors.map(v => ({
      id: v.id,
      values: v.values,
      metadata: v.metadata
    }))
  };
  await index.upsert({ upsertRequest });
}
